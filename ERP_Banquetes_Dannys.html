Visor SQL

El SQL real que genera cada libro contable — esto es lo que hace un ERP empresarial

🗂️ Estructura de Tablas (DDL)
-- TABLA 1: Catálogo de Cuentas CREATE TABLE CatalogoCuentas ( CodigoCuenta TEXT PRIMARY KEY, NombreCuenta TEXT NOT NULL, Naturaleza TEXT NOT NULL, -- 'D'=Deudora, 'A'=Acreedora Tipo TEXT NOT NULL, -- 'Clase','Grupo','Mayor' Nivel INTEGER NOT NULL -- 1,2,3 ); -- TABLA 2: Encabezado de cada asiento CREATE TABLE Partidas ( IDPartida INTEGER PRIMARY KEY AUTOINCREMENT, NumeroPartida INTEGER NOT NULL, Fecha TEXT NOT NULL, Concepto TEXT NOT NULL, EsAjuste INTEGER DEFAULT 0 ); -- TABLA 3: Cada línea de Debe/Haber CREATE TABLE DetallePartidas ( IDDetalle INTEGER PRIMARY KEY AUTOINCREMENT, IDPartida INTEGER REFERENCES Partidas(IDPartida), CodigoCuenta TEXT REFERENCES CatalogoCuentas(CodigoCuenta), Descripcion TEXT, Debe REAL DEFAULT 0, Haber REAL DEFAULT 0 );
📖 Query — Libro Diario
SELECT p.NumeroPartida, p.Fecha, p.Concepto, d.CodigoCuenta, c.NombreCuenta, d.Descripcion, d.Debe, d.Haber FROM Partidas p JOIN DetallePartidas d ON p.IDPartida = d.IDPartida JOIN CatalogoCuentas c ON d.CodigoCuenta = c.CodigoCuenta ORDER BY p.Fecha, p.NumeroPartida;
📚 Query — Libro Mayor (GROUP BY)
SELECT c.CodigoCuenta, c.NombreCuenta, c.Naturaleza, SUM(d.Debe) AS TotalDebe, SUM(d.Haber) AS TotalHaber, CASE WHEN c.Naturaleza = 'D' THEN SUM(d.Debe) - SUM(d.Haber) ELSE SUM(d.Haber) - SUM(d.Debe) END AS Saldo FROM CatalogoCuentas c JOIN DetallePartidas d ON c.CodigoCuenta = d.CodigoCuenta WHERE c.Nivel = 3 GROUP BY c.CodigoCuenta HAVING SUM(d.Debe) > 0 OR SUM(d.Haber) > 0 ORDER BY c.CodigoCuenta;
